\item ({\bf 10 points}) Convert the following linear program into canonical form.
\[
\begin{array}{ll}
  \max & 4 x_1 + 2x_2 - 7 x_3 \\
  \mbox{s.t.} & 2 x_1 -  x_2 + 4 x_3 \leq 18 \\
  &  4 x_1 + 2 x_2 \geq 10 \\
  & x_1  \geq 0, x_2 \leq 0, x_3 \text{ free}
\end{array}
\]

\end{enumerate}
